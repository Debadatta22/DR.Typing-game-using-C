/* ── Box / Display Helpers ───────────────────────────────── */

/* Print a full-width horizontal rule:  +====...====+  or  +----...----+ */
static void hRule(char corner, char fill) {
    putchar(corner);
    for (int i = 0; i < BOX_WIDTH - 2; i++) putchar(fill);
    putchar(corner);
    putchar('\n');
}

/* Print one box row:   | <text padded to INNER_WIDTH> |               */
static void boxRow(const char *text) {
    printf("| %-*s |\n", INNER_WIDTH, text);
}

/* Print one centred box row */
static void boxRowCentre(const char *text) {
    int len = (int)strlen(text);
    int left  = (INNER_WIDTH - len) / 2;
    int right = INNER_WIDTH - len - left;
    printf("|%*s%s%*s|\n", left + 1, "", text, right + 1, "");
}

/* ── displayTextInBox ────────────────────────────────────────
   Word-wraps 'text' to INNER_WIDTH columns and prints each
   wrapped line inside a "| ... |" box.
   Works for texts of any length (15 s short or 150 s long).
───────────────────────────────────────────────────────────── */
void displayTextInBox(const char *text) {
    hRule('+', '-');

    /* Make a local copy so we can tokenise */
    char buf[MAX_TEXT_LENGTH];
    strncpy(buf, text, MAX_TEXT_LENGTH - 1);
    buf[MAX_TEXT_LENGTH - 1] = '\0';

    char line[MAX_TEXT_LENGTH] = "";
    int  lineLen = 0;

    /* Split on spaces; when a word would exceed INNER_WIDTH, flush the line */
    char *word = strtok(buf, " ");
    while (word != NULL) {
        int wlen = (int)strlen(word);

        if (lineLen == 0) {
            /* First word on this line */
            strcat(line, word);
            lineLen += wlen;
        } else if (lineLen + 1 + wlen <= INNER_WIDTH) {
            /* Word fits with one space */
            strcat(line, " ");
            strcat(line, word);
            lineLen += 1 + wlen;
        } else {
            /* Flush current line and start a new one */
            boxRow(line);
            strcpy(line, word);
            lineLen = wlen;
        }
        word = strtok(NULL, " ");
    }
    /* Flush the last line */
    if (lineLen > 0) boxRow(line);

    hRule('+', '-');
}

/* ── printHeading ────────────────────────────────────────── */
void printHeading(const char *text) {
    printf("\n");
    hRule(' ', '=');
    boxRowCentre(text);
    hRule(' ', '=');
    printf("\n");
}

/* ── File I/O ────────────────────────────────────────────── */
void loadUsers() {
    FILE *f = fopen("users.txt", "r");
    if (!f) return;
    while (userCount < MAX_USERS &&
           fscanf(f, "%49s %49s",
                  users[userCount].username,
                  users[userCount].password) == 2)
        userCount++;
    fclose(f);
}

void saveUsers() {
    FILE *f = fopen("users.txt", "w");
    if (!f) { printf(">> Error saving users!\n"); return; }
    for (int i = 0; i < userCount; i++)
        fprintf(f, "%s %s\n", users[i].username, users[i].password);
    fclose(f);
}

void loadRecords() {
    FILE *f = fopen("records.txt", "r");
    if (!f) return;
    while (recordCount < MAX_RECORDS) {
        float cons; long dt;
        if (fscanf(f, "%49s %d %d %d %d %d %f %ld",
                   records[recordCount].username,
                   &records[recordCount].wpm,
                   &records[recordCount].accuracy,
                   &records[recordCount].mistakes,
                   &records[recordCount].raw,
                   &records[recordCount].charsTyped,
                   &cons, &dt) != 8) break;
        records[recordCount].consistency = cons;
        records[recordCount].date        = (time_t)dt;
        recordCount++;
    }
    fclose(f);
}

void saveRecords() {
    FILE *f = fopen("records.txt", "w");
    if (!f) { printf("Error saving records!\n"); return; }
    for (int i = 0; i < recordCount; i++)
        fprintf(f, "%s %d %d %d %d %d %.2f %ld\n",
                records[i].username, records[i].wpm, records[i].accuracy,
                records[i].mistakes, records[i].raw, records[i].charsTyped,
                records[i].consistency, (long)records[i].date);
    fclose(f);
}

/* ── login / register ────────────────────────────────────── */
int login() {
    int choice;
    scanf("%d", &choice);
    getchar();

    if (choice == 1) {
        char username[50], password[50];
        printf("Enter username: "); fgets(username, 50, stdin);
        username[strcspn(username, "\n")] = 0;
        printf("Enter password: "); fgets(password, 50, stdin);
        password[strcspn(password, "\n")] = 0;

        for (int i = 0; i < userCount; i++) {
            if (strcmp(username, users[i].username) == 0 &&
                strcmp(password, users[i].password) == 0) {
                strcpy(currentUser, username);
                return 1;
            }
        }
        printf("Invalid username or password!\n");
        myDelay(2000);
        return 0;

    } else if (choice == 2) {
        registerUser();
        return 1;
    }

    printf("Invalid choice!\n");
    myDelay(2000);
    return 0;
}

void registerUser() {
    if (userCount >= MAX_USERS) {
        printf("Maximum user limit reached!\n"); myDelay(2000); return;
    }
    char username[50], password[50];
    printf("Enter new username: "); fgets(username, 50, stdin);
    username[strcspn(username, "\n")] = 0;

    for (int i = 0; i < userCount; i++) {
        if (strcmp(username, users[i].username) == 0) {
            printf("Username already exists!\n"); myDelay(2000); return;
        }
    }
    printf("Enter password: "); fgets(password, 50, stdin);
    password[strcspn(password, "\n")] = 0;

    strcpy(users[userCount].username, username);
    strcpy(users[userCount].password, password);
    userCount++;
    strcpy(currentUser, username);
    printf("Registration successful!\n");
    myDelay(2000);
}

/* ── typingTest ──────────────────────────────────────────── */
void typingTest(int duration) {
    clearScreen();

    char filename[30];
    sprintf(filename, "text_%d.txt", duration);

    FILE *f = fopen(filename, "r");
    if (!f) { printf("Error opening text file!\n"); myDelay(2000); return; }

    /* Count lines */
    int lineCount = 0;
    char ch;
    while ((ch = (char)fgetc(f)) != EOF)
        if (ch == '\n') lineCount++;

    if (lineCount == 0) {
        printf("No text available!\n"); fclose(f); myDelay(2000); return;
    }

    /* Pick a random line */
    rewind(f);
    srand((unsigned)time(NULL));
    int sel = rand() % lineCount;

    char line[MAX_LINE_LENGTH];
    for (int i = 0; i <= sel; i++) {
        if (fgets(line, MAX_LINE_LENGTH, f) == NULL) {
            printf("Error reading text file!\n"); fclose(f); myDelay(2000); return;
        }
    }
    fclose(f);
    line[strcspn(line, "\n")] = 0;   /* strip trailing newline */

    /* ── Show the text ── */
    printf("\n  .:~*~:.  Typing Test  .:~*~:.\n\n");
    printf("  Type the following text exactly:\n\n");
    displayTextInBox(line);
    printf("\n  Press any key to start ...");
    _getch();

    clearScreen();
    printf("\n  .:~*~:.  Typing Test  .:~*~:.\n\n");
    printf("  Type the following text exactly:\n\n");
    displayTextInBox(line);
    printf("\n  Time: %d sec  |  Press ENTER when done\n\n", duration);
    printf("  > ");

    time_t startTime = time(NULL);
    time_t endTime   = startTime + duration;

    char  input[MAX_TEXT_LENGTH] = {0};
    int   pos        = 0;
    int   charsTyped = 0;

    while (time(NULL) < endTime) {
        long remaining = (long)(endTime - time(NULL));
        /* Redraw timer on same line */
        printf("\r  [%3ld sec left]  > %s   ", remaining, input);
        fflush(stdout);

        if (_kbhit()) {
            char c = (char)_getch();
            if (c == '\r') break;
            else if (c == '\b') {
                if (pos > 0) { pos--; input[pos] = '\0'; }
            } else if (isprint((unsigned char)c) && pos < MAX_TEXT_LENGTH - 1) {
                input[pos++] = c;
                input[pos]   = '\0';
                charsTyped++;
            }
        }
        Sleep(40);
    }
    printf("\n");

    /* ── Score calculation ── */
    int totalChars  = (int)strlen(line);
    int inputLen    = (int)strlen(input);
    int correctChars = 0;

    for (int i = 0; i < inputLen && i < totalChars; i++)
        if (input[i] == line[i]) correctChars++;

    int mistakes = inputLen - correctChars;
    if (mistakes < 0) mistakes = 0;

    float timeMin = (float)duration / 60.0f;
    int   raw     = (timeMin > 0) ? (int)((charsTyped / 5.0f) / timeMin) : 0;
    float consist = (charsTyped > 0) ? ((float)correctChars / charsTyped) * 100.0f : 0.0f;

    showResults(correctChars, totalChars, mistakes, duration, raw, charsTyped, consist);
}

/* ── showResults ─────────────────────────────────────────── */
void showResults(int correctChars, int totalChars, int mistakes,
                 int duration, int raw, int charsTyped, float consistency) {
    clearScreen();

    float accuracy = (totalChars > 0) ? ((float)correctChars / totalChars) * 100.0f : 0.0f;
    int   wpm      = (duration  > 0)  ? (int)((correctChars / 5.0f) / (duration / 60.0f)) : 0;

    printHeading(".:*  Typing Test Results  *:.");

    /* Results box */
    printf("  +------------------------------------------------+\n");
    printf("  |  %-20s   %20s  |\n", "Metric", "Value");
    printf("  +------------------------------------------------+\n");
    printf("  |  Correct characters  :  %-20d |\n", correctChars);
    printf("  |  Total characters    :  %-20d |\n", totalChars);
    printf("  |  Mistakes            :  %-20d |\n", mistakes);
    printf("  |  Accuracy            :  %-18.2f %% |\n", accuracy);
    printf("  |  Words Per Minute    :  %-20d |\n", wpm);
    printf("  |  Raw WPM             :  %-20d |\n", raw);
    printf("  |  Characters Typed    :  %-20d |\n", charsTyped);
    printf("  |  Consistency         :  %-18.2f %% |\n", consistency);
    printf("  |  Test Duration       :  %-17d sec |\n", duration);
    printf("  +------------------------------------------------+\n\n");

    /* Rating */
    const char *rating =
        wpm >= 80 ? ".:* Legendary Typist *:." :
        wpm >= 60 ? ".:* Expert Typist *:."     :
        wpm >= 40 ? ".:* Good Typist *:."        :
        wpm >= 20 ? ".:* Average Typist *:."     :
                    ".:* Keep Practising! *:.";
    printf("  Performance: %s\n\n", rating);

    /* Save record */
    if (recordCount < MAX_RECORDS) {
        strcpy(records[recordCount].username,  currentUser);
        records[recordCount].wpm         = wpm;
        records[recordCount].accuracy    = (int)accuracy;
        records[recordCount].mistakes    = mistakes;
        records[recordCount].raw         = raw;
        records[recordCount].charsTyped  = charsTyped;
        records[recordCount].consistency = consistency;
        records[recordCount].date        = time(NULL);
        recordCount++;
        saveRecords();   /* persist immediately */
    }

    printf("  Press any key to continue ...");
    _getch();
}

/* ── showScoreboard ──────────────────────────────────────── */
void showScoreboard() {
    clearScreen();

    /* Bubble-sort descending by WPM */
    for (int i = 0; i < recordCount - 1; i++)
        for (int j = 0; j < recordCount - i - 1; j++)
            if (records[j].wpm < records[j+1].wpm) {
                ScoreRecord tmp = records[j];
                records[j] = records[j+1];
                records[j+1] = tmp;
            }

    int show = (recordCount > 10) ? 10 : recordCount;
